#1286

Submultimi1

Cerință

Se citește un număr natural nenul n. Să se afişeze, în ordine lexicografică, toate submulțimile nevide ale mulţimii {1,2,..,n} pentru care diferența dintre oricare două elemente este mai mare decât 1.

Fişierul de intrare submultimi1.in conţine pe prima linie numărul n.

Fişierul de ieşire submultimi1.out va conţine pe fiecare linie câte o submulțime, elementele unei submulțimi fiind separate printr-un spațiu.

Exemplu

Input

5

Output

1 
1 3 
1 3 5 
1 4 
1 5 
2 
2 4 
2 5 
3 
3 5 
4 
5
Restricţii şi precizări

* 1 ≤ n ≤ 10
* elementele fiecărei submulțimi vor fi afișate în ordine crescătoare

Cod (soluție)
solutie.cpp

#include <fstream>
using namespace std;
ifstream cin("submultimi1.in");
ofstream cout("submultimi1.out");
int n,sol[11];

void init(int k)
{
    sol[k]=0;
}
int succesor(int k)
{
    if (sol[k]<n)
    {
        sol[k]++;
        return 1;
    }
    return 0;
}
int valid(int k)
{
    for (int i=1;i<k;i++)
        if(sol[k]==sol[i])
            return 0;

    if(k>1 & sol[k]-sol[k-1]<=1)
        return 0;
    if (k>1 & sol[k]<sol[k-1])
        return 0;
    return 1;
}
void tipar(int k)
{
    for (int i=1; i<=k;i++)
        cout<<sol[i]<< " ";
    cout << endl;
}
void back(int k)
{
    init(k);
    while(succesor(k))
    {
        if (valid(k))
        {
            tipar(k);
            back(k+1);
        }
    }
}
int main()
{
    cin>>n;
    back(1);
    return 0;
}

PartitiiNr

Cerință

Se dă un număr natural n. Determinaţi, în ordine lexicografică, toate modalităţile de a-l scrie pe n ca sumă de numere naturale ordonate crescător astfel încât diferența dintre doi termeni consecutivi ai sumei să fie cel puțin 2.

Fişierul de intrare partitiinr.in conţine pe prima linie numărul n.

Fişierul de ieşire partitiinr.out va conţine pe pe fiecare linie câte un şir de numere naturale ordonate crescător, separate prin câte un spaţiu. Suma numerelor din fiecare şir este n, iar diferența dintre oricare doi termeni consecutivi este cel puțin 2. Şirurile vor fi afişate în ordine lexicografică.

Exemplu

Input

15

Output

1 3 11 
1 4 10 
1 5 9 
1 6 8 
1 14 
2 4 9 
2 5 8 
2 13 
3 5 7 
3 12 
4 11 
5 10 
6 9 
15
Restricţii şi precizări

* 1 ≤ n ≤ 100

Cod (soluție)
solutie.cpp

#include <iostream>
#include <fstream>
using namespace std;
ifstream f("partitiinr.in");
ofstream c("partitiinr.out");
int n,sol[101],i;
void init(int k)
{
    sol[k]=0;
}
int succesor(int k)
{
    if(sol[k]<n)
    {
        sol[k]++;
        return 1;
    }
    else
        return 0;
}
int valid(int k)
{
    if(k>1&&sol[k]-sol[k-1]<2) return 0;
    if (k>1&&sol[k]<sol[k-1]) return 0;
    return 1;
}
int solutie(int k)
{
    int s=0;
    for(i=1;i<=k;i++)
        s+=sol[i];
    return s;
}
void tipar(int k)
{
    for(i=1;i<=k;i++)
        c<<sol[i]<<' ';
    c<<endl;
}
void back(int k)
{
    init(k);
    while (succesor(k))
    {
        if (valid(k))
        {
            if (solutie(k)==n)
                tipar(k);
            else if (solutie(k)<n)
                back(k+1);
        }

    }
}
int main()
{
    f>>n;
    back(1);
    return 0;
}

#344

Paranteze

Cerință

Se dă un număr natural par n. Generați toate șirurile de n paranteze rotunde care se închid corect.

Fişierul de intrare paranteze.in conţine pe prima linie numărul n.

Fişierul de ieşire paranteze.out va conţine pe fiecare linie câte un șir de n paranteze rotunde care se închid corect. Șirurile vor fi afișate în ordine lexicografică, considerând paranteza deschisa ( mai mică decât paranteza închisă ).

Exemplu

Input

6

Output

((()))
(()())
(())()
()(())
()()()
Restricţii şi precizări

* 1 ≤ n ≤ 20, număr natural par

Cod (soluție)
solutie.cpp

#include <iostream>
#include <fstream>
using namespace std;
ifstream f("paranteze.in");
ofstream g("paranteze.out");
char s[21],d ='(',i =')';
int n,m;
void back(int k,int cd,int ci)
{
    if(k==n)
    {
        for(int j=0;j<n;j++)
            g<<s[j];
        g<<endl;
        return;
    }
   
    if (cd<m)
    {
        s[k]=d;
        back(k+1,cd+1,ci);
    }
    
    if (ci<cd)
    {
        s[k]=i;
        back(k+1,cd,ci+1);
    }
}
int main()
{
    f>>n;
    m=n/2;
    back(0,0,0);
    return 0;
}

Cuvinte11

Cerință

Se citește un cuvânt S format din litere mici distincte ale alfabetului englez și un număr m. Afișați în ordine lexicografică toate cuvintele formate din cel puțin m litere ordonate alfabetic din S, în care nu există vocale alăturate și se termină cu consoană.

Programul citește de la tastatură cuvântul S și numărul m.

Programul va afișa pe ecran cuvintele cerute, câte unul pe un rând.

Exemplu

Input

barni 3

Output

abin
abinr
abir
abn
abnr
abr
anr
bin
binr
bir
bnr
inr
Restricții și precizări

* 1 < m ≤ lungimea lui S ≤ 10

Cod (soluție)
solutie.cpp

#include <iostream>
#include <cstring>
using namespace std;
char v[11],s[11];
int n;
int vocala(char d)
{
    if(strchr("aeiou",d)!=0)
        return 1;
    return 0;
}
int valid(int k)
{
    if(k==0)
        return 1;
    if(vocala(s[k])!=0 && vocala(s[k-1])!=0)
        return 0;
    return 1;
}
void tipar(int k)
{
    for(int i=0;i<=k;i++)
        cout<<s[i];
    cout<<endl;
}
void back(int k,int c)
{
    for(int i=c;i<strlen(v);i++)
    {
        s[k]=v[i];
        if(valid(k))
        {
            if(k+1>=n && vocala(s[k])==0)
                tipar(k);
            back(k+1,i+1);
        }
    }
}
int main()
{
    cin>>v>>n;
    for(int i=0;i<strlen(v);i++)
        for(int j=i+1;j<strlen(v);j++)
            if(v[i]>v[j])
                swap(v[i],v[j]);
    back(0,0);
    return 0;
}
